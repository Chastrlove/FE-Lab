# 二叉树最大路径

```ts
const findMaxPath = (node) => {
  let maxValue = 0;
  function findMax(node) {
    if (!node) {
      return 0;
    }
    const leftValue = Math.max(findMax(node.left), 0);
    const rightValue = Math.max(findMax(node.right), 0);
    //选择最大路径
    const maxPathValue = node.value + Math.max(leftValue, rightValue);
    //考虑以当前节点为路径根的情况
    const rootPathValue = node.value + leftValue + rightValue;
    maxValue = Math.max(...[maxValue, maxPathValue, rootPathValue]);
    return maxPathValue;
  }
  findMax(node);
  return maxValue;
};

const root = {
  value: -10,
  left: {
    value: 9,
  },
  right: {
    left: {
      value: 15,
    },
    right: {
      value: 7,
    },
    value: 20,
  },
};

const d = findMaxPath(root);

console.log(d);
```
